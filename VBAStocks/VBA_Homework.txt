Sub Ticker()

For Each ws In Worksheets

    Dim Ticker As String
    Dim lastrow
    Dim Ticker_Summary_Table_Row As Integer
    Dim Opening_Price As Double
    Dim Closing_Price As Double
    Dim Date_min As Double
    Dim Yearly_Change As Double
    Dim Percent_Change As Double
    Dim Total_Stock_Volume As Double
    
    lastrow = ws.Cells(Rows.Count, 1).End(xlUp).Row
    
    Ticker_Summary_Table_Row = 2
    Total_Stock_Volume = 0
    
    ws.Range("I1").Value = "Ticker"
    ws.Range("J1").Value = "Yearly Change"
    ws.Range("K1").Value = "Percent Change"
    ws.Range("L1").Value = "Total Stock Volume"
    
    For i = 2 To lastrow
    
        If ws.Cells(i + 1, 1) <> ws.Cells(i, 1).Value Then
        
            Ticker = ws.Cells(i, 1).Value
            ws.Range("I" & Ticker_Summary_Table_Row).Value = Ticker
            ws.Columns("I:I").EntireColumn.AutoFit
            
            Closing_Price = ws.Cells(i, 6).Value
            
            Yearly_Change = Closing_Price - Opening_Price
            ws.Range("J" & Ticker_Summary_Table_Row).Value = Yearly_Change
            ws.Columns("J:J").EntireColumn.AutoFit
            
            Percent_Change = Yearly_Change / Opening_Price
            ws.Range("K" & Ticker_Summary_Table_Row).Value = Percent_Change
            ws.Range("K" & Ticker_Summary_Table_Row).NumberFormat = "0.00%"
            ws.Columns("K:K").EntireColumn.AutoFit
            
            Total_Stock_Volume = Total_Stock_Volume + ws.Cells(i, 7).Value
            ws.Range("L" & Ticker_Summary_Table_Row).Value = Total_Stock_Volume
            ws.Columns("L:L").EntireColumn.AutoFit
            
            Ticker_Summary_Table_Row = Ticker_Summary_Table_Row + 1
            
            Total_Stock_Volume = 0
            
        Else
        
            Total_Stock_Volume = Total_Stock_Volume + ws.Cells(i, 7).Value
            
            If ws.Cells(i, 2).Value < ws.Cells(i - 1, 2).Value Then
            
                Opening_Price = ws.Cells(i, 3).Value
                
            End If
        End If
    
    Next i
    
    Next ws
                
End Sub

Sub colorindex()

For Each ws In Worksheets

Dim lastrowsummary As Integer
    
    lastrowsummary = ws.Cells(Rows.Count, 10).End(xlUp).Row
    
    For j = 2 To lastrowsummary
    
    If ws.Cells(j, 10).Value > 0 Then
    
        ws.Cells(j, 10).Interior.colorindex = 4
        
    Else
            
        ws.Cells(j, 10).Interior.colorindex = 3
        
    End If
        
    Next j
    
    Next ws
    
End Sub

Sub summarytable()

For Each ws In Worksheets

    Dim Greatest_Increase_Ticker As String
    Dim Greatest_Increase_Value As Double
    Dim Greatest_Decrease_Ticker As String
    Dim Greatest_Decrease_Value As Double
    Dim Greatest_Volume_Ticker_ As String
    Dim Greatest_Volume_Value As Double
    
    Dim lastrowsummary As Integer
    
    lastrowsummary = ws.Cells(Rows.Count, 11).End(xlUp).Row
    
    ws.Range("O2").Value = "Greatest % Increase"
    ws.Columns("O:O").EntireColumn.AutoFit
    ws.Range("O3").Value = "Greatest % Decrease"
    ws.Range("O4").Value = "Greatest Total Volume"
    ws.Range("P1").Value = "Ticker"
    ws.Columns("P:P").EntireColumn.AutoFit
    ws.Range("Q1").Value = "Value"
    ws.Columns("Q:Q").EntireColumn.AutoFit
    
    Greatest_Increase_Value = ws.Application.WorksheetFunction.Max(ws.Range("K:K"))
    ws.Range("Q2").Value = Greatest_Increase_Value
    ws.Range("Q2").NumberFormat = "0.00%"
    
    Greatest_Decrease_Value = ws.Application.WorksheetFunction.Min(ws.Range("K:K"))
    ws.Range("Q3").Value = Greatest_Decrease_Value
    ws.Range("Q3").NumberFormat = "0.00%"
    
    Greatest_Volume_Value = ws.Application.WorksheetFunction.Max(ws.Range("L:L"))
    ws.Range("Q4").Value = Greatest_Volume_Value
    
    For j = 2 To lastrowsummary
    
        If ws.Range("Q2").Value = ws.Cells(j, 11).Value Then
        Greatest_Increase_Ticker = ws.Cells(j, 9).Value
        ws.Range("P2").Value = Greatest_Increase_Ticker
        
        ElseIf ws.Range("Q3").Value = Cells(j, 11).Value Then
            
            Greatest_Decrease_Ticker = ws.Cells(j, 9).Value
            ws.Range("P3").Value = Greatest_Decrease_Ticker
            
            ElseIf ws.Range("Q4").Value = Cells(j, 12).Value Then
            
                Greatest_Volume_Ticker = ws.Cells(j, 9).Value
                ws.Range("P4").Value = Greatest_Volume_Ticker
                
            End If
        
        Next j
        
    Next ws


End Sub
